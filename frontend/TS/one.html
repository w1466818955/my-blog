# 一、基础类型

基础类型: string、number、boolean、null、undefined以及ES6的Symbol和ES10的BigInt
1
// 字符串
let str: string = 'hello world';

// 数字
let num: number = 123;

// 布尔
let bool: boolean = true;

// null
let n: null = null;

// undefined
let u: undefined = undefined;

// Symbol
let s: symbol = Symbol('key');

// BigInt
let bi: bigint = BigInt(123);
