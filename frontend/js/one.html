# 1.for...of 无法循环遍历对象

let userMsg = {
  0: "nick",
  1: "ff",
  2: "dd",
  3: "gg",
};

for (let key in userMsg) {
  console.log(key, userMsg[key]);
}
// 输出结果

1 ff
